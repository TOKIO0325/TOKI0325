body { margin: 0; height: 100vh; background: #111; color: #fff; display: flex; justify-content: center; align-items: center; font-family: "Segoe UI", sans-serif; } .clock { text-align: center; } #time { font-size: 64px; letter-spacing: 2px; } #date { margin-top: 10px; font-size: 18px; color: #aaa; } function updateClock() { const now = new Date(); const hours = String(now.getHours()).padStart(2, "0"); const minutes = String(now.getMinutes()).padStart(2, "0"); const seconds = String(now.getSeconds()).padStart(2, "0"); const year = now.getFullYear(); const month = String(now.getMonth() + 1).padStart(2, "0"); const day = String(now.getDate()).padStart(2, "0"); document.getElementById("time").textContent = `${hours}:${minutes}:${seconds}`; document.getElementById("date").textContent = `${year}-${month}-${day}`; } updateClock(); setInterval(updateClock, 1000);
